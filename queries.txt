--players to be sold due to low number of participation in matches
SELECT TOP 200 NAME, APEARANCE,GOALS
FROM PLAYER_PERFORMANCE
WHERE APEARANCE < (SELECT AVG(APEARANCE) FROM PLAYER_PERFORMANCE)
ORDER BY APEARANCE ASC

--top player based on goals , apperance
SELECT TOP 1 NAME, GOALS, APEARANCE, PENALTIES, YELLOW_C, RED_C
FROM PLAYER_PERFORMANCE
where APEARANCE!=0
ORDER BY ((APEARANCE/38)*(GOALS)) DESC;

--Top Scoring Team:
SELECT TOP 1 WITH TIES TNAME AS TopScoringTeam, GF AS GoalsScored
FROM TEAM_PERFORMANCE
ORDER BY GF DESC;

--Top losses Team:
SELECT TOP 1 WITH TIES TNAME AS TopLostTeam, L
FROM TEAM_PERFORMANCE
ORDER BY L DESC;


--Top players got red cards
SELECT TOP 1 WITH TIES NAME , RED_C
FROM PLAYER_PERFORMANCE
ORDER BY RED_C DESC;
